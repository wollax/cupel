    ```csharp
    [MemoryDiagnoser]
    public class PipelineBenchmark
    {
        private CupelPipeline _pipeline = null!;
        private ContextItem[] _items = null!;

        [Params(100, 250, 500)]
        public int ItemCount { get; set; }

        [GlobalSetup]
        public void Setup()
        {
            // Build pipeline with realistic configuration:
            // - CompositeScorer with RecencyScorer (weight=2) + PriorityScorer (weight=1)
            // - GreedySlice (default)
            // - UShapedPlacer (to exercise the more complex placer)
            // - Budget: TargetTokens = ItemCount * 5 (roughly half the items), MaxTokens = ItemCount * 10
            // - Dedup enabled (default)

            var budget = new ContextBudget(
                maxTokens: ItemCount * 10,
                targetTokens: ItemCount * 5);

            _pipeline = CupelPipeline.CreateBuilder()
                .WithBudget(budget)
                .AddScorer(new RecencyScorer(), 2.0)
                .AddScorer(new PriorityScorer(), 1.0)
                .WithPlacer(new UShapedPlacer())
                .Build();

            // Generate items with realistic variety
            var baseTime = DateTimeOffset.UtcNow;
            _items = new ContextItem[ItemCount];
            for (var i = 0; i < ItemCount; i++)
            {
                _items[i] = new ContextItem
                {
                    Content = $"Context item number {i} with some realistic content length",
                    Tokens = 8 + (i % 5),  // 8-12 tokens, varied
                    Kind = i % 3 == 0 ? ContextKind.ToolOutput : ContextKind.Message,
                    Priority = i % 7 == 0 ? i : null,
                    Timestamp = baseTime.AddMinutes(-ItemCount + i),
                    Pinned = i == 0  // One pinned item
                };
            }
        }

        [Benchmark]
        public ContextResult FullPipeline()
        {
            return _pipeline.Execute(_items);
        }

        [Benchmark]
        public ContextResult FullPipelineWithTracing()
        {
            var trace = new DiagnosticTraceCollector();
            return _pipeline.Execute(_items, trace);
        }
    }
    ```

    Key points:
    - FullPipeline uses default NullTraceCollector (should show zero Gen0 from trace paths)
    - FullPipelineWithTracing exercises the tracing path for comparison
    - Items have mixed kinds, timestamps, priorities, one pinned item, some duplicate potential
    - Budget is set so slicer must actually select (not just pass everything through)
    - Uses UShapedPlacer to exercise the more complex placement algorithm
    - Uses AddScorer (CompositeScorer path) for realistic multi-signal scoring

    Import the necessary namespaces: Wollax.Cupel, Wollax.Cupel.Scoring, Wollax.Cupel.Diagnostics, BenchmarkDotNet.Attributes.
